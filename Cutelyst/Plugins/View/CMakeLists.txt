if (PLUGIN_VIEW_GRANTLEE)
    find_package(Grantlee5 REQUIRED)
    message(STATUS "PLUGIN: View::Grantlee, enabled.")
    set (GRANTLEE_PLUGINS_DIR "${CMAKE_INSTALL_LIBDIR}/grantlee/${Grantlee5_VERSION_MAJOR}.${Grantlee5_VERSION_MINOR}" CACHE PATH "Output directory for grantlee plugins")
    add_subdirectory(Grantlee)
endif (PLUGIN_VIEW_GRANTLEE)

if (PLUGIN_VIEW_CLEARSILVER)
    find_library(ClearSilver_CS   libneo_cs.a)
    find_library(ClearSilver_Util libneo_utl.a)
    if (ClearSilver_CS AND ClearSilver_Util)
        message(STATUS "PLUGIN: View::ClearSilver, enabled.")
        add_subdirectory(ClearSilver)
    else ()
        message(FATAL_ERROR "ClearSilver libraries not found.")
    endif ()
endif (PLUGIN_VIEW_CLEARSILVER)

add_subdirectory(JSON)

if (PLUGIN_VIEW_EMAIL)
    find_package(SimpleMail2Qt5 2.0.0 REQUIRED)
    message(STATUS "PLUGIN: View::Email, enabled.")
    add_subdirectory(Email)
endif (PLUGIN_VIEW_EMAIL)
